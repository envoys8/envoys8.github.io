```abap
DATA lSLAYOUT TYPE LVC_S_LAYO.
DATA: LCGRID TYPE REF TO CL_GUI_ALV_GRID.

CALL FUNCTION 'GET_GLOBALS_FROM_SLVC_FULLSCR'
IMPORTING
E_GRID = LCGRID.

CALL METHOD LCGRID->CHECK_CHANGED_DATA.
CALL METHOD LCGRID->GET_FRONTEND_LAYOUT
IMPORTING
ES_LAYOUT = lSLAYOUT.
lSLAYOUT-CWIDTH_OPT = 'X'.
CALL METHOD LCGRID->SET_FRONTEND_LAYOUT
EXPORTING
IS_LAYOUT = lSLAYOUT.
```
